60 static std::vector<meshkernel::Point> ConvertGeometryListToPointVector(const GeometryList& geometryListIn)
61 {
62 std::vector<meshkernel::Point> result;
63 if (geometryListIn.num_coordinates == 0)
64 {
65 return result;
66 }
67 result.reserve(geometryListIn.num_coordinates);
68 result.emplace_back(geometryListIn.coordinates_x[0], geometryListIn.coordinates_y[0]);
69 // remove consecutive duplicated point leading to 0 edge length
70 for (auto i = 1; i < geometryListIn.num_coordinates; ++i)
71 {
72 if (meshkernel::IsEqual(geometryListIn.coordinates_x[i], result.back().x) &&
73 meshkernel::IsEqual(geometryListIn.coordinates_y[i], result.back().y))
74 {
75 continue;
76 }
77 result.emplace_back(geometryListIn.coordinates_x[i], geometryListIn.coordinates_y[i]);
78 }
79 return result;
80 }

86 static std::vector<std::vector<meshkernel::Point>> ConvertGeometryListToVectorOfPointVectors(GeometryList const& geometryListIn)
87 {
88 std::vector<std::vector<meshkernel::Point>> result;
89 std::vector<meshkernel::Point> chunk;
90 chunk.reserve(geometryListIn.num_coordinates);
91 for (auto i = 0; i < geometryListIn.num_coordinates; ++i)
92 {
93
94 if (meshkernel::IsEqual(geometryListIn.coordinates_x[i], geometryListIn.geometry_separator))
95 {
96 result.emplace_back(chunk);
97 chunk.clear();
98 }
99 else
100 {
101 chunk.emplace_back(geometryListIn.coordinates_x[i], geometryListIn.coordinates_y[i]);
102 }
103 }
104
105 if (!chunk.empty())
106 {
107 result.emplace_back(chunk);
108 }
109
110 return result;
111 }

117 template <typename T>
118 static std::vector<std::vector<T>> ConvertVectorToVectorOfVectors(const std::vector<T> input, T separator)
119 {
120 std::vector<std::vector<T>> result;
121 std::vector<T> chunk;
122 for (size_t i = 0; i < input.size(); ++i)
123 {
124 if (meshkernel::IsEqual(input[i], separator))
125 {
126 result.emplace_back(chunk);
127 chunk.clear();
128 }
129 else
130 {
131 chunk.emplace_back(input[i]);
132 }
133 }
134 if (!chunk.empty())
135 {
136 result.emplace_back(chunk);
137 }
138 return result;

145 static std::vector<bool> ConvertIntegerArrayToBoolVector(const int inputArray[], size_t inputSize)
146 {
147 std::vector<bool> result(inputSize);
148 for (size_t i = 0; i < inputSize; ++i)
149 {
150 switch (inputArray[i])
151 {
152 case 0:
153 result[i] = false;
154 break;
155 case 1:
156 result[i] = true;
157 break;
158 default:
159 throw std::invalid_argument("MeshKernel: Invalid 1D mask.");
160 }
161 }
162 return result;
163 }

168 static std::vector<meshkernel::Sample> ConvertGeometryListToSampleVector(const GeometryList& geometryListIn)
169 {
170 if (geometryListIn.num_coordinates == 0)
171 {
172 throw std::invalid_argument("MeshKernel: The samples are empty.");
173 }
174 std::vector<meshkernel::Sample> result;
175 result.reserve(geometryListIn.num_coordinates);

188 {
189 if (pointVector.size() < static_cast<size_t>(result.num_coordinates))

205 static void ConvertSampleVectorToGeometryList(std::vector<meshkernel::Point> const& valuesCoordinates, std::vector<double> const& values, GeometryList& result)
206 {
207 if (valuesCoordinates.size() != values.size())
208 {
209 throw std::invalid_argument("MeshKernel: The size of the valuesCoordinates-vector is not equal to the size of the values-vector");
210 }
211
212 if (values.size() < static_cast<size_t>(result.num_coordinates))
213 {
214 throw std::invalid_argument("MeshKernel: Invalid memory allocation, the value-vector size is smaller than the number of coordinates in the result vector.");
215 }
216
217 for (auto i = 0; i < result.num_coordinates; ++i)
218 {
219 result.coordinates_x[i] = valuesCoordinates[i].x;
220 result.coordinates_y[i] = valuesCoordinates[i].y;
221 result.values[i] = values[i];
222 }

228 template <meshkernel::InterpolatableType T>
229 static std::vector<meshkernel::Sample> ComputeGriddedDataSamples(const GriddedSamples& griddedSamples)
230 {
231 std::vector<meshkernel::Sample> result;
232 meshkernel::Point origin{griddedSamples.x_origin, griddedSamples.y_origin};
233 const auto numSamples = static_cast<size_t>(griddedSamples.num_x * griddedSamples.num_y);
234 result.resize(numSamples);
235 const T* valuePtr = static_cast<T*>(griddedSamples.values);
236 if (griddedSamples.x_coordinates == nullptr || griddedSamples.y_coordinates == nullptr)
237 {
238 meshkernel::UInt index = 0;
239
240 for (int j = 0; j < griddedSamples.num_x; ++j)
241 {
242 for (int i = griddedSamples.num_y - 1; i >= 0; --i)
243 {
244 const auto griddedIndex = griddedSamples.num_x * i + j;
245 result[index].x = origin.x + j * griddedSamples.cell_size;
246 result[index].y = origin.y + i * griddedSamples.cell_size;
247 result[index].value = static_cast<double>(valuePtr[griddedIndex]);
248 index++;
249 }
250 }
251 return result;
252 }
253
254 meshkernel::UInt index = 0;
255 for (int j = 0; j < griddedSamples.num_x; ++j)
256 {
257 for (int i = griddedSamples.num_y - 1; i >= 0; --i)
258 {
259 const auto griddedIndex = griddedSamples.num_x * i + j;
260 result[index].x = origin.x + griddedSamples.x_coordinates[griddedIndex];
261 result[index].y = origin.y + griddedSamples.y_coordinates[griddedIndex];
262 result[index].value = static_cast<double>(valuePtr[griddedIndex]);
263 index++;
264 }
265 }
266 return result;
267 }

272 static std::vector<meshkernel::Sample> ConvertGriddedData(const GriddedSamples& griddedSamples)
273 {
274 std::vector<meshkernel::Sample> result;
275 if (griddedSamples.num_x <= 0 || griddedSamples.num_y <= 0)
276 {
277 return result;
278 }
279
280 if (griddedSamples.value_type == static_cast<int>(meshkernel::InterpolationDataTypes::Short))
281 {
282 return ComputeGriddedDataSamples<short>(griddedSamples);
283 }
284 if (griddedSamples.value_type == static_cast<int>(meshkernel::InterpolationDataTypes::Float))
285 {
286 return ComputeGriddedDataSamples<float>(griddedSamples);
287 }
288 if (griddedSamples.value_type == static_cast<int>(meshkernel::InterpolationDataTypes::Double))
289 {
290 return ComputeGriddedDataSamples<double>(griddedSamples);
291 }
292 if (griddedSamples.value_type == static_cast<int>(meshkernel::InterpolationDataTypes::Int))
293 {
294 return ComputeGriddedDataSamples<int>(griddedSamples);
295 }
296 throw meshkernel::MeshKernelError("The value type for the gridded data samples is invalid.");
297 }

302 static void SetSplines(const GeometryList& geometryListIn, meshkernel::Splines& spline)
303 {
304 if (geometryListIn.num_coordinates == 0)
305 {
306 return;
307 }
308
309 const auto splineCornerPoints = ConvertGeometryListToPointVector(geometryListIn);
310
311 const auto indices = FindIndices(splineCornerPoints,
312 0,
313 static_cast<meshkernel::UInt>(splineCornerPoints.size()),
314 meshkernel::constants::missing::doubleValue);
315
316 for (const auto& index : indices)
317 {
318 const auto& [startIndex, endIndex] = index;
319 const auto size = endIndex - startIndex + 1;
320 if (size > 0)
321 {
322 spline.AddSpline(splineCornerPoints, startIndex, size);
323 }
324 }

330 static void SetMesh2dApiDimensions(const meshkernel::Mesh& mesh2d, Mesh2D& mesh2dApi)
331 {

341 mesh2dApi.num_valid_nodes = static_cast<int>(mesh2d.GetNumValidNodes());
342 mesh2dApi.num_edges = static_cast<int>(mesh2d.GetNumEdges());
343 mesh2dApi.num_valid_edges = static_cast<int>(mesh2d.GetNumValidEdges());

349 static void SetMesh2dApiNodeEdgeData(meshkernel::Mesh2D& mesh2d, Mesh2D& mesh2dApi)
350 {
351 if (mesh2dApi.num_nodes != static_cast<int>(mesh2d.GetNumNodes()))
352 {
353 throw meshkernel::ConstraintError("The number of nodes in the mesh2d api structure does not equal the number of nodes in the grid, {} /= {}",
354 mesh2dApi.num_nodes, mesh2d.GetNumNodes());
355 }
356
357 if (mesh2dApi.num_edges != static_cast<int>(mesh2d.GetNumEdges()))
358 {
359 throw meshkernel::ConstraintError("The number of edges in the mesh2d api structure does not equal the number of edges in the grid, {} /= {}",
360 mesh2dApi.num_edges, mesh2d.GetNumEdges());
361 }
362
363 for (meshkernel::UInt n = 0; n < mesh2d.GetNumNodes(); ++n)
364 {
365 mesh2dApi.node_x[n] = mesh2d.Node(n).x;
366 mesh2dApi.node_y[n] = mesh2d.Node(n).y;
367 }
368
369 for (meshkernel::UInt edgeIndex = 0; edgeIndex < mesh2d.GetNumEdges(); ++edgeIndex)
370 {
371 mesh2dApi.edge_nodes[edgeIndex * 2] = static_cast<int>(mesh2d.GetEdge(edgeIndex).first);
372 mesh2dApi.edge_nodes[edgeIndex * 2 + 1] = static_cast<int>(mesh2d.GetEdge(edgeIndex).second);
373 }
374 SetMesh2dApiDimensions(mesh2d, mesh2dApi);
375 }

380 static void SetMesh2dApiData(meshkernel::Mesh2D& mesh2d, Mesh2D& mesh2dApi)
381 {
382 std::vector<meshkernel::Point> edgeCentres = meshkernel::MeshEdgeCenters::Compute(mesh2d);
383
384 for (meshkernel::UInt n = 0; n < mesh2d.GetNumNodes(); n++)
385 {
386 mesh2dApi.node_x[n] = mesh2d.Node(n).x;
387 mesh2dApi.node_y[n] = mesh2d.Node(n).y;
388 }
389
390 for (meshkernel::UInt edgeIndex = 0; edgeIndex < mesh2d.GetNumEdges(); edgeIndex++)
391 {
392 mesh2dApi.edge_x[edgeIndex] = edgeCentres[edgeIndex].x;
393 mesh2dApi.edge_y[edgeIndex] = edgeCentres[edgeIndex].y;
394 mesh2dApi.edge_nodes[edgeIndex * 2] = static_cast<int>(mesh2d.GetEdge(edgeIndex).first);
395 mesh2dApi.edge_nodes[edgeIndex * 2 + 1] = static_cast<int>(mesh2d.GetEdge(edgeIndex).second);
396
397 const auto& firstEdgeFace = mesh2d.m_edgesFaces[edgeIndex][0];
398 mesh2dApi.edge_faces[edgeIndex * 2] = firstEdgeFace == meshkernel::constants::missing::uintValue ? -1 : static_cast<int>(firstEdgeFace);
399 const auto& secondEdgeFace = mesh2d.m_edgesFaces[edgeIndex][1];
400 mesh2dApi.edge_faces[edgeIndex * 2 + 1] = secondEdgeFace == meshkernel::constants::missing::uintValue ? -1 : static_cast<int>(secondEdgeFace);
401 }
402
403 int faceIndex = 0;
404 for (size_t f = 0; f < mesh2d.GetNumFaces(); f++)
405 {
406 mesh2dApi.face_x[f] = mesh2d.m_facesMassCenters[f].x;
407 mesh2dApi.face_y[f] = mesh2d.m_facesMassCenters[f].y;
408 mesh2dApi.nodes_per_face[f] = static_cast<int>(mesh2d.m_facesNodes[f].size());
409 for (size_t n = 0; n < mesh2d.m_facesNodes[f].size(); ++n)
410 {
411 mesh2dApi.face_nodes[faceIndex] = static_cast<int>(mesh2d.m_facesNodes[f][n]);
412 mesh2dApi.face_edges[faceIndex] = static_cast<int>(mesh2d.m_facesEdges[f][n]);
413 faceIndex++;
414 }
415 }
416 SetMesh2dApiDimensions(mesh2d, mesh2dApi);
417 }

430

435 }
436
437 int count = 0;
438
439 for (meshkernel::UInt n = 0; n < curvilinearGrid.NumN(); n++)
440 {
441 for (meshkernel::UInt m = 0; m < curvilinearGrid.NumM(); m++)
442 {
443 curvilinearGridApi.node_x[count] = curvilinearGrid.GetNode(n, m).x;
444 curvilinearGridApi.node_y[count] = curvilinearGrid.GetNode(n, m).y;
445 ++count;
446 }

546 template <meshkernel::InterpolatableType T>
547 static std::unique_ptr<meshkernel::MeshInterpolation> CreateBilinearInterpolator(const meshkernel::Mesh2D& mesh2d,
548 const GriddedSamples& griddedSamples)
549 {
550 meshkernel::Point origin{griddedSamples.x_origin, griddedSamples.y_origin};
551 if (griddedSamples.x_coordinates == nullptr || griddedSamples.y_coordinates == nullptr)
552 {
553 return std::make_unique<meshkernel::BilinearInterpolationOnGriddedSamples<T>>(mesh2d,
554 griddedSamples.num_x,
555 griddedSamples.num_y,
556 origin,
557 griddedSamples.cell_size,
558 std::span<T const>{reinterpret_cast<T const* const>(griddedSamples.values),
559 static_cast<size_t>(griddedSamples.num_x * griddedSamples.num_y)});
560 }
561 return std::make_unique<meshkernel::BilinearInterpolationOnGriddedSamples<T>>(mesh2d,
562 std::span<double const>{griddedSamples.x_coordinates,
563 static_cast<size_t>(griddedSamples.num_x)},
564 std::span<double const>{griddedSamples.y_coordinates,
565 static_cast<size_t>(griddedSamples.num_y)},
566 std::span<T const>{reinterpret_cast<T const* const>(griddedSamples.values),
567 static_cast<size_t>(griddedSamples.num_x * griddedSamples.num_y)});
568 }

570 static std::unique_ptr<meshkernel::MeshInterpolation> CreateBilinearInterpolatorBasedOnType(const GriddedSamples& griddedSamples,
571 const meshkernel::Mesh2D& mesh2d)
572 {
573
574 switch (static_cast<meshkernel::InterpolationDataTypes>(griddedSamples.value_type))

584 default:
585 throw meshkernel::MeshKernelError("Invalid value_type for GriddedSamples");
586 }

589 static void FillFacePolygons(const meshkernel::Mesh2D& mesh2d,
590 const std::vector<bool>& facesInPolygon,
591 const GeometryList& facePolygons)
592 {
593 meshkernel::UInt count = 0;
594 for (meshkernel::UInt f = 0; f < mesh2d.GetNumFaces(); ++f)
595 {
596 if (!facesInPolygon[f])
597 {
598 continue;
599 }
600
601 const auto& faceNodes = mesh2d.m_facesNodes[f];
602 if (count != 0)
603 {
604 facePolygons.coordinates_x[count] = meshkernel::constants::missing::doubleValue;
605 facePolygons.coordinates_y[count] = meshkernel::constants::missing::doubleValue;
606 count++;
607 }
608
609 for (meshkernel::UInt n = 0u; n < faceNodes.size(); ++n)
610 {
611 const auto& currentNode = mesh2d.Node(faceNodes[n]);
612 facePolygons.coordinates_x[count] = currentNode.x;
613 facePolygons.coordinates_y[count] = currentNode.y;
614 count++;
615 }
616 const auto& currentNode = mesh2d.Node(faceNodes[0]);
617 facePolygons.coordinates_x[count] = currentNode.x;
618 facePolygons.coordinates_y[count] = currentNode.y;
619 count++;
620 }
621 }
