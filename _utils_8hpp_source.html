59  static std::vector<meshkernel::Point> ConvertGeometryListToPointVector(const GeometryList& geometryListIn)
60  {
61  std::vector<meshkernel::Point> result;
62  if (geometryListIn.num_coordinates == 0)
63  {
64  return result;
65  }
66  result.reserve(geometryListIn.num_coordinates);
67  result.emplace_back(geometryListIn.coordinates_x[0], geometryListIn.coordinates_y[0]);
68  // remove consecutive duplicated point leading to 0 edge length
69  for (auto i = 1; i < geometryListIn.num_coordinates; ++i)
70  {
71  if (meshkernel::IsEqual(geometryListIn.coordinates_x[i], result.back().x) &&
72  meshkernel::IsEqual(geometryListIn.coordinates_y[i], result.back().y))
73  {
74  continue;
75  }
76  result.emplace_back(geometryListIn.coordinates_x[i], geometryListIn.coordinates_y[i]);
77  }
78  return result;
79  }

85  static std::vector<std::vector<meshkernel::Point>> ConvertGeometryListToVectorOfPointVectors(GeometryList const& geometryListIn)
86  {
87  std::vector<std::vector<meshkernel::Point>> result;
88  std::vector<meshkernel::Point> chunk;
89  chunk.reserve(geometryListIn.num_coordinates);
90  for (auto i = 0; i < geometryListIn.num_coordinates; ++i)
91  {
92 
93  if (meshkernel::IsEqual(geometryListIn.coordinates_x[i], geometryListIn.geometry_separator))
94  {
95  result.emplace_back(chunk);
96  chunk.clear();
97  }
98  else
99  {
100  chunk.emplace_back(geometryListIn.coordinates_x[i], geometryListIn.coordinates_y[i]);
101  }
102  }
103 
104  if (!chunk.empty())
105  {
106  result.emplace_back(chunk);
107  }
108 
109  return result;
110  }

227  template <meshkernel::InterpolatableType T>
228  static std::vector<meshkernel::Sample> ComputeGriddedDataSamples(const GriddedSamples& griddedSamples)
229  {
230  std::vector<meshkernel::Sample> result;
231  meshkernel::Point origin{griddedSamples.x_origin, griddedSamples.y_origin};
232  const auto numSamples = static_cast<size_t>(griddedSamples.num_x * griddedSamples.num_y);
233  result.resize(numSamples);
234  const T* valuePtr = static_cast<T*>(griddedSamples.values);
235  if (griddedSamples.x_coordinates == nullptr || griddedSamples.y_coordinates == nullptr)
236  {
237  meshkernel::UInt index = 0;
238 
239  for (int j = 0; j < griddedSamples.num_x; ++j)
240  {
241  for (int i = griddedSamples.num_y - 1; i >= 0; --i)
242  {
243  const auto griddedIndex = griddedSamples.num_x * i + j;
244  result[index].x = origin.x + j * griddedSamples.cell_size;
245  result[index].y = origin.y + i * griddedSamples.cell_size;
246  result[index].value = static_cast<double>(valuePtr[griddedIndex]);
247  index++;
248  }
249  }
250  return result;
251  }
252 
253  meshkernel::UInt index = 0;
254  for (int j = 0; j < griddedSamples.num_x; ++j)
255  {
256  for (int i = griddedSamples.num_y - 1; i >= 0; --i)
257  {
258  const auto griddedIndex = griddedSamples.num_x * i + j;
259  result[index].x = origin.x + griddedSamples.x_coordinates[griddedIndex];
260  result[index].y = origin.y + griddedSamples.y_coordinates[griddedIndex];
261  result[index].value = static_cast<double>(valuePtr[griddedIndex]);
262  index++;
263  }
264  }
265  return result;

302  {
303  if (geometryListIn.num_coordinates == 0)
304  {
305  return;

308  const auto splineCornerPoints = ConvertGeometryListToPointVector(geometryListIn);

313  meshkernel::constants::missing::doubleValue);
314 
315  for (const auto& index : indices)

319  if (size > 0)

321  spline.AddSpline(splineCornerPoints, startIndex, size);

356  if (mesh2dApi.num_edges != static_cast<int>(mesh2d.GetNumEdges()))

366  }
367 
368  for (meshkernel::UInt edgeIndex = 0; edgeIndex < mesh2d.GetNumEdges(); ++edgeIndex)
369  {
370  mesh2dApi.edge_nodes[edgeIndex * 2] = static_cast<int>(mesh2d.GetEdge(edgeIndex).first);
371  mesh2dApi.edge_nodes[edgeIndex * 2 + 1] = static_cast<int>(mesh2d.GetEdge(edgeIndex).second);
372  }

379  static void SetMesh2dApiData(meshkernel::Mesh2D& mesh2d, Mesh2D& mesh2dApi)
380  {
381  mesh2d.ComputeEdgesCenters();
382  for (meshkernel::UInt n = 0; n < mesh2d.GetNumNodes(); n++)
383  {
384  mesh2dApi.node_x[n] = mesh2d.Node(n).x;
385  mesh2dApi.node_y[n] = mesh2d.Node(n).y;
386  }
387 
388  for (meshkernel::UInt edgeIndex = 0; edgeIndex < mesh2d.GetNumEdges(); edgeIndex++)
389  {
390  mesh2dApi.edge_x[edgeIndex] = mesh2d.m_edgesCenters[edgeIndex].x;
391  mesh2dApi.edge_y[edgeIndex] = mesh2d.m_edgesCenters[edgeIndex].y;
392  mesh2dApi.edge_nodes[edgeIndex * 2] = static_cast<int>(mesh2d.GetEdge(edgeIndex).first);
393  mesh2dApi.edge_nodes[edgeIndex * 2 + 1] = static_cast<int>(mesh2d.GetEdge(edgeIndex).second);
394 
395  const auto& firstEdgeFace = mesh2d.m_edgesFaces[edgeIndex][0];
396  mesh2dApi.edge_faces[edgeIndex * 2] = firstEdgeFace == meshkernel::constants::missing::uintValue ? -1 : static_cast<int>(firstEdgeFace);
397  const auto& secondEdgeFace = mesh2d.m_edgesFaces[edgeIndex][1];
398  mesh2dApi.edge_faces[edgeIndex * 2 + 1] = secondEdgeFace == meshkernel::constants::missing::uintValue ? -1 : static_cast<int>(secondEdgeFace);
399  }
400 
401  int faceIndex = 0;
402  for (size_t f = 0; f < mesh2d.GetNumFaces(); f++)
403  {
404  mesh2dApi.face_x[f] = mesh2d.m_facesMassCenters[f].x;
405  mesh2dApi.face_y[f] = mesh2d.m_facesMassCenters[f].y;
406  mesh2dApi.nodes_per_face[f] = static_cast<int>(mesh2d.m_facesNodes[f].size());
407  for (size_t n = 0; n < mesh2d.m_facesNodes[f].size(); ++n)
408  {
409  mesh2dApi.face_nodes[faceIndex] = static_cast<int>(mesh2d.m_facesNodes[f][n]);
410  mesh2dApi.face_edges[faceIndex] = static_cast<int>(mesh2d.m_facesEdges[f][n]);
411  faceIndex++;
412  }
413  }
414  SetMesh2dApiDimensions(mesh2d, mesh2dApi);
415  }

587  static void FillFacePolygons(const meshkernel::Mesh2D& mesh2d,
588  const std::vector<bool>& facesInPolygon,
589  const GeometryList& facePolygons)
590  {
591  meshkernel::UInt count = 0;
592  for (meshkernel::UInt f = 0; f < mesh2d.GetNumFaces(); ++f)
593  {
594  if (!facesInPolygon[f])
595  {
596  continue;
597  }
598 
599  const auto& faceNodes = mesh2d.m_facesNodes[f];
600  if (count != 0)
601  {
602  facePolygons.coordinates_x[count] = meshkernel::constants::missing::doubleValue;
603  facePolygons.coordinates_y[count] = meshkernel::constants::missing::doubleValue;
604  count++;
605  }
606 
607  for (meshkernel::UInt n = 0u; n < faceNodes.size(); ++n)
608  {
609  const auto& currentNode = mesh2d.Node(faceNodes[n]);
610  facePolygons.coordinates_x[count] = currentNode.x;
611  facePolygons.coordinates_y[count] = currentNode.y;
612  count++;
613  }
614  const auto& currentNode = mesh2d.Node(faceNodes[0]);
615  facePolygons.coordinates_x[count] = currentNode.x;
616  facePolygons.coordinates_y[count] = currentNode.y;
