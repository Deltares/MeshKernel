148  static void SetMesh2d(std::shared_ptr<meshkernel::Mesh2D> mesh2d, Mesh2D& mesh2dApi)
149  {
150  mesh2d->ComputeEdgesCenters();
151  for (auto n = 0; n < mesh2d->GetNumNodes(); n++)
152  {
153  mesh2dApi.node_x[n] = mesh2d->m_nodes[n].x;
154  mesh2dApi.node_y[n] = mesh2d->m_nodes[n].y;
155  }
156 
157  for (auto edgeIndex = 0; edgeIndex < mesh2d->GetNumEdges(); edgeIndex++)
158  {
159  mesh2dApi.edge_x[edgeIndex] = mesh2d->m_edgesCenters[edgeIndex].x;
160  mesh2dApi.edge_y[edgeIndex] = mesh2d->m_edgesCenters[edgeIndex].y;
161  mesh2dApi.edge_nodes[edgeIndex * 2] = static_cast<int>(mesh2d->m_edges[edgeIndex].first);
162  mesh2dApi.edge_nodes[edgeIndex * 2 + 1] = static_cast<int>(mesh2d->m_edges[edgeIndex].second);
163  }
164 
165  int faceIndex = 0;
166  for (auto f = 0; f < mesh2d->GetNumFaces(); f++)
167  {
168  mesh2dApi.face_x[f] = mesh2d->m_facesMassCenters[f].x;
169  mesh2dApi.face_y[f] = mesh2d->m_facesMassCenters[f].y;
170  mesh2dApi.nodes_per_face[f] = static_cast<int>(mesh2d->m_facesNodes[f].size());
171  for (auto n = 0; n < mesh2d->m_facesNodes[f].size(); ++n)
172  {
173  mesh2dApi.face_nodes[faceIndex] = static_cast<int>(mesh2d->m_facesNodes[f][n]);
174  faceIndex++;
175  }
176  }
177  }

226  static std::vector<meshkernel::Point> ComputeLocations(const Mesh2D& mesh2d,
227  meshkernel::MeshLocations interpolationLocation)
228  {
229  std::vector<meshkernel::Point> locations;
230  if (interpolationLocation == meshkernel::MeshLocations::Nodes)
231  {
232  locations = meshkernel::ConvertToNodesVector(mesh2d.num_nodes, mesh2d.node_x, mesh2d.node_y);
233  }
234  if (interpolationLocation == meshkernel::MeshLocations::Edges)
235  {
236  const auto edges = meshkernel::ConvertToEdgeNodesVector(mesh2d.num_edges, mesh2d.edge_nodes);
237  const auto nodes = meshkernel::ConvertToNodesVector(mesh2d.num_nodes, mesh2d.node_x, mesh2d.node_y);
238  locations = ComputeEdgeCenters(nodes, edges);
239  }
240  if (interpolationLocation == meshkernel::MeshLocations::Faces)
241  {
242  locations = meshkernel::ConvertToFaceCentersVector(mesh2d.num_faces, mesh2d.face_x, mesh2d.face_y);
243  }
244 
245  return locations;
246  }
