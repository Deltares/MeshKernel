56  static std::vector<meshkernel::Point> ConvertGeometryListToPointVector(const GeometryList& geometryListIn)
57  {
58  std::vector<meshkernel::Point> result;
59  if (geometryListIn.num_coordinates == 0)
60  {
61  return result;
62  }
63  result.reserve(geometryListIn.num_coordinates);
64  result.emplace_back(geometryListIn.coordinates_x[0], geometryListIn.coordinates_y[0]);
65  // remove consecutive duplicated point leading to 0 edge length
66  for (auto i = 1; i < geometryListIn.num_coordinates; ++i)
67  {
68  if (meshkernel::IsEqual(geometryListIn.coordinates_x[i], result.back().x) &&
69  meshkernel::IsEqual(geometryListIn.coordinates_y[i], result.back().y))
70  {
71  continue;
72  }
73  result.emplace_back(geometryListIn.coordinates_x[i], geometryListIn.coordinates_y[i]);
74  }
75  return result;
76  }

82  static std::vector<std::vector<meshkernel::Point>> ConvertGeometryListToVectorOfPointVectors(GeometryList const& geometryListIn)
83  {
84  std::vector<std::vector<meshkernel::Point>> result;
85  std::vector<meshkernel::Point> chunk;
86  chunk.reserve(geometryListIn.num_coordinates);
87  for (auto i = 0; i < geometryListIn.num_coordinates; ++i)
88  {
89 
90  if (meshkernel::IsEqual(geometryListIn.coordinates_x[i], geometryListIn.geometry_separator))
91  {
92  result.emplace_back(chunk);
93  chunk.clear();
94  }
95  else
96  {
97  chunk.emplace_back(geometryListIn.coordinates_x[i], geometryListIn.coordinates_y[i]);
98  }
99  }
100 
101  if (!chunk.empty())
102  {
103  result.emplace_back(chunk);
104  }
105 
106  return result;
107  }

224  template <meshkernel::InterpolatableType T>
225  static std::vector<meshkernel::Sample> ComputeGriddedDataSamples(const GriddedSamples& griddedSamples)
226  {
227  std::vector<meshkernel::Sample> result;
228  meshkernel::Point origin{griddedSamples.x_origin, griddedSamples.y_origin};
229  const auto numSamples = static_cast<size_t>(griddedSamples.num_x * griddedSamples.num_y);
230  result.resize(numSamples);
231  const T* valuePtr = static_cast<T*>(griddedSamples.values);
232  if (griddedSamples.x_coordinates == nullptr || griddedSamples.y_coordinates == nullptr)
233  {
234  meshkernel::UInt index = 0;
235 
236  for (int j = 0; j < griddedSamples.num_x; ++j)
237  {
238  for (int i = griddedSamples.num_y - 1; i >= 0; --i)
239  {
240  const auto griddedIndex = griddedSamples.num_x * i + j;
241  result[index].x = origin.x + j * griddedSamples.cell_size;
242  result[index].y = origin.y + i * griddedSamples.cell_size;
243  result[index].value = static_cast<double>(valuePtr[griddedIndex]);
244  index++;
245  }
246  }
247  return result;
248  }
249 
250  meshkernel::UInt index = 0;
251  for (int j = 0; j < griddedSamples.num_x; ++j)
252  {
253  for (int i = griddedSamples.num_y - 1; i >= 0; --i)
254  {
255  const auto griddedIndex = griddedSamples.num_x * i + j;
256  result[index].x = origin.x + griddedSamples.x_coordinates[griddedIndex];
257  result[index].y = origin.y + griddedSamples.y_coordinates[griddedIndex];
258  result[index].value = static_cast<double>(valuePtr[griddedIndex]);
259  index++;
260  }
261  }
262  return result;

335  static void SetMesh2dApiData(meshkernel::Mesh2D& mesh2d, Mesh2D& mesh2dApi)
336  {
337  mesh2d.ComputeEdgesCenters();
338  for (size_t n = 0; n < mesh2d.GetNumNodes(); n++)
339  {
340  mesh2dApi.node_x[n] = mesh2d.m_nodes[n].x;
341  mesh2dApi.node_y[n] = mesh2d.m_nodes[n].y;
342  }
343 
344  for (size_t edgeIndex = 0; edgeIndex < mesh2d.GetNumEdges(); edgeIndex++)
345  {
346  mesh2dApi.edge_x[edgeIndex] = mesh2d.m_edgesCenters[edgeIndex].x;
347  mesh2dApi.edge_y[edgeIndex] = mesh2d.m_edgesCenters[edgeIndex].y;
348  mesh2dApi.edge_nodes[edgeIndex * 2] = static_cast<int>(mesh2d.m_edges[edgeIndex].first);
349  mesh2dApi.edge_nodes[edgeIndex * 2 + 1] = static_cast<int>(mesh2d.m_edges[edgeIndex].second);
350 
351  const auto& firstEdgeFace = mesh2d.m_edgesFaces[edgeIndex][0];
352  mesh2dApi.edge_faces[edgeIndex * 2] = firstEdgeFace == meshkernel::constants::missing::uintValue ? -1 : static_cast<int>(firstEdgeFace);
353  const auto& secondEdgeFace = mesh2d.m_edgesFaces[edgeIndex][1];
354  mesh2dApi.edge_faces[edgeIndex * 2 + 1] = secondEdgeFace == meshkernel::constants::missing::uintValue ? -1 : static_cast<int>(secondEdgeFace);
355  }
356 
357  int faceIndex = 0;
358  for (size_t f = 0; f < mesh2d.GetNumFaces(); f++)
359  {
360  mesh2dApi.face_x[f] = mesh2d.m_facesMassCenters[f].x;
361  mesh2dApi.face_y[f] = mesh2d.m_facesMassCenters[f].y;
362  mesh2dApi.nodes_per_face[f] = static_cast<int>(mesh2d.m_facesNodes[f].size());
363  for (size_t n = 0; n < mesh2d.m_facesNodes[f].size(); ++n)
364  {
365  mesh2dApi.face_nodes[faceIndex] = static_cast<int>(mesh2d.m_facesNodes[f][n]);
366  mesh2dApi.face_edges[faceIndex] = static_cast<int>(mesh2d.m_facesEdges[f][n]);
367  faceIndex++;
368  }
369  }
370  }
