233  static Point Rotate(const Point& point, const double angle, const Point& reference)
234  {
235  const auto translatedPoint = point - reference;
236 
237  const auto angleInRad = angle * constants::conversion::degToRad;
238  const auto cosineAngle = std::cos(angleInRad);
239  const auto sinAngle = std::sin(angleInRad);
240  Point result(translatedPoint.x * cosineAngle - translatedPoint.y * sinAngle,
241  translatedPoint.x * sinAngle + translatedPoint.y * cosineAngle);
242 
243  return result + reference;
244  }
