187 template <typename T>
188 void meshkernel::BoundingBox::Reset(const std::vector<T>& points, size_t start, size_t end)
189 {
190  double minx = std::numeric_limits<double>::max();
191  double maxx = std::numeric_limits<double>::lowest();
192  double miny = std::numeric_limits<double>::max();
193  double maxy = std::numeric_limits<double>::lowest();
194 
195  for (size_t i = start; i <= end; ++i)
196  {
197  const auto& point = points[i];
198 
199  if (point.IsValid())
200  {
201  minx = std::min(minx, point.x);
202  maxx = std::max(maxx, point.x);
203  miny = std::min(miny, point.y);
204  maxy = std::max(maxy, point.y);
205  }
206  }
207  m_lowerLeft = Point(minx, miny);
208  m_upperRight = Point(maxx, maxy);
209 }
