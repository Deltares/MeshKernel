161 inline const meshkernel::PolygonalEnclosure& meshkernel::Polygons::Enclosure(const UInt index) const
162 {
163  if (IsEmpty())
164  {
165  throw ConstraintError("Enclosures list is empty.");
166  }
167 
168  if (index >= m_enclosures.size())
169  {
170  throw ConstraintError("Invalid enclosure index: {}, maximum index: {}", index, m_enclosures.size() - 1);
171  }
172 
173  return m_enclosures[index];
174 }
