80  template <std::derived_from<Point> T>
81  void Compute(const std::vector<T>& inputNodes,
82  TriangulationOptions triangulationOption,
83  double averageTriangleArea,
84  UInt estimatedNumberOfTriangles)
85  {
86  std::vector<double> xLocalPolygon(inputNodes.size());
87  std::vector<double> yLocalPolygon(inputNodes.size());
88  for (UInt i = 0; i < inputNodes.size(); ++i)
89  {
90  xLocalPolygon[i] = inputNodes[i].x;
91  yLocalPolygon[i] = inputNodes[i].y;
92  }
93 
94  m_numFaces = -1;
95  m_numEdges = 0;
96  m_numNodes = 0;
97 
98  int numInputNodes = static_cast<int>(inputNodes.size());
99  auto intTriangulationOption = static_cast<int>(triangulationOption);
100 
101  if (estimatedNumberOfTriangles == 0)
102  {
103  estimatedNumberOfTriangles = static_cast<UInt>(inputNodes.size()) * 6 + 10;
104  }
105 
106  // If the number of estimated triangles is not sufficient, triangulation must be repeated
107  while (m_numFaces < 0)
108  {
109  m_numFaces = static_cast<int>(estimatedNumberOfTriangles);
110 
111  m_faceNodesFlat.resize(estimatedNumberOfTriangles * 3);
112  std::ranges::fill(m_faceNodesFlat, 0);
113 
114  m_edgeNodesFlat.resize(estimatedNumberOfTriangles * 2);
115  std::ranges::fill(m_edgeNodesFlat, 0);
116 
117  m_faceEdgesFlat.resize(estimatedNumberOfTriangles * 3);
118  std::ranges::fill(m_faceEdgesFlat, 0);
119 
120  m_xCoordFlat.resize(estimatedNumberOfTriangles * 3, constants::missing::doubleValue);
121  std::ranges::fill(m_xCoordFlat, 0.0);
122 
123  m_yCoordFlat.resize(estimatedNumberOfTriangles * 3, constants::missing::doubleValue);
124  std::ranges::fill(m_yCoordFlat, 0.0);
125 
126  Triangulation(intTriangulationOption,
127  xLocalPolygon.data(),
128  yLocalPolygon.data(),
129  numInputNodes,
130  m_faceNodesFlat.data(), // INDX
131  &m_numFaces,
132  m_edgeNodesFlat.data(), // EDGEINDX
133  &m_numEdges,
134  m_faceEdgesFlat.data(), // TRIEDGE
135  m_xCoordFlat.data(),
136  m_yCoordFlat.data(),
137  &m_numNodes,
138  averageTriangleArea);
139  if (estimatedNumberOfTriangles > 0)
140  {
141  estimatedNumberOfTriangles = -m_numFaces;
142  }
143  }
144  }
