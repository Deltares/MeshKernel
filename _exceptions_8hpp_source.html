142  class MeshKernelError : public std::exception
143  {
144  public:
148  template <typename... Args>
149  MeshKernelError(FormatString const& format_string, Args&&... args)
150  : m_what(),
151  m_source_location(format_string.SourceLocation())
152  {
153  if (sizeof...(args) == 0)
154  {
155  m_formatted_message = format_string.String();
156  }
157  else
158  {
159  m_formatted_message = fmt_ns::vformat(format_string.String(), fmt_ns::make_format_args(args...));
160  }
161  }
162 
164  virtual ~MeshKernelError() = default;
165 
168  const char* what() const noexcept override
169  {
170 #if HAVE_SRC_LOC_IN_ERR_MSGS
171  m_what = fmt_ns::format("Exception of type '{}' in {} ({}:{}) {}: {}\n",
172  Category().Name(),
173  StrippedFilePath(),
174  m_source_location.line(),
175  m_source_location.column(),
176  m_source_location.function_name(),
177  FormattedMessage());
178 #else
179  m_what = fmt_ns::format("Exception of type '{}': {}\n",
180  Category().Name(),
181  FormattedMessage());
182 #endif
183  return m_what.c_str();
184  }
185 
188  [[nodiscard]] ExitCode Code() const { return Category().Code(); }
189 
190  protected:
193  [[nodiscard]] virtual ErrorCategory Category() const
194  {
195  return {"MeshKernelError", ExitCode::MeshKernelErrorCode};
196  }
197 
199  [[nodiscard]] virtual std::string FormattedMessage() const { return m_formatted_message; }
200 
201  private:
202  std::string m_formatted_message;
203 
204  mutable std::string m_what;
205 
206  std::source_location m_source_location;
207 
210  [[nodiscard]] std::string StrippedFilePath() const
211  {
212  std::string path = m_source_location.file_name();
213 #ifdef CMAKE_SRC_DIR
214  std::string path_to_erase(TO_STR_LITERAL(CMAKE_SRC_DIR));
215 #ifdef _WIN32
216  std::replace(path_to_erase.begin(), path_to_erase.end(), '/', '\\');
217 #endif
218  if (size_t pos = path.find(path_to_erase); pos != std::string::npos)
219  {
220  // erase including the trailing slash
221  path.erase(pos, path_to_erase.length() + 1);
222 #ifdef _WIN32
223  std::replace(path.begin(), path.end(), '\\', '/');
224 #endif
225  }
226 #endif
227  return path;
228  }
229  };
