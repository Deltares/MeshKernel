46  {

49  using LongLat = bg::model::d2::point_xy<double, bg::cs::geographic<bg::degree>>;

52  using UTM = bg::model::d2::point_xy<double, Projection>;

55  explicit ConvertSphericalToCartesianBase(const ProjectionConversion& proj) : m_projection(proj) {}
56

59 
61  Projection SourceProjection() const
62  {
63  return Projection::spherical;
64  }
65 
67  Projection TargetProjection() const
68  {
69  return Projection::cartesian;
70  }
71 
73  Point operator()(const Point& pnt) const
74  {
75  LongLat longLat(pnt.x, pnt.y);
76  UTM utm{0.0, 0.0};

86  };
