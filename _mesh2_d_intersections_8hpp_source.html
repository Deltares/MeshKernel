135  static void updateEdgeIntersections(const UInt segmentIndex,
136  const UInt edgeIndex,
137  const Edge edge,
138  const std::vector<double>& cumulativeLength,
139  const double crossProductValue,
140  const double adimensionalEdgeDistance,
141  const double adimensionalPolylineSegmentDistance,
142  std::vector<EdgeMeshPolyLineIntersection>& intersections)
143  {
144  const auto [edgeFirstNode, edgeSecondNode] = edge;
145 
146  intersections[edgeIndex].polylineSegmentIndex = static_cast<int>(segmentIndex);
147  intersections[edgeIndex].polylineDistance = cumulativeLength[segmentIndex] +
148  adimensionalPolylineSegmentDistance * (cumulativeLength[segmentIndex + 1] - cumulativeLength[segmentIndex]);
149  intersections[edgeIndex].adimensionalPolylineSegmentDistance = adimensionalPolylineSegmentDistance;
150  intersections[edgeIndex].edgeFirstNode = crossProductValue < 0 ? edgeSecondNode : edgeFirstNode;
151  intersections[edgeIndex].edgeSecondNode = crossProductValue < 0 ? edgeFirstNode : edgeSecondNode;
152  intersections[edgeIndex].edgeDistance = adimensionalEdgeDistance;
153  intersections[edgeIndex].edgeIndex = edgeIndex;
154  }
