145  template <InterpolatableType T>
146  void BilinearInterpolationOnGriddedSamples<T>::Compute()
147  {
148  const auto numNodes = m_mesh.GetNumNodes();
149  const auto numEdges = m_mesh.GetNumEdges();
150  const auto numFaces = m_mesh.GetNumFaces();
151 
152  m_nodeResults.resize(numNodes);
153  std::ranges::fill(m_nodeResults, constants::missing::doubleValue);
154  for (UInt n = 0; n < numNodes; ++n)
155  {
156  const auto node = m_mesh.m_nodes[n];
157  m_nodeResults[n] = Interpolation(node);
158  }
159 
160  m_edgeResults.resize(numEdges);
161  std::ranges::fill(m_edgeResults, constants::missing::doubleValue);
162  for (UInt e = 0; e < numEdges; ++e)
163  {
164  const auto& [first, second] = m_mesh.m_edges[e];
165  m_edgeResults[e] = 0.5 * (m_nodeResults[first] + m_nodeResults[second]);
166  }
167 
168  m_faceResults.resize(numFaces, constants::missing::doubleValue);
169  std::ranges::fill(m_faceResults, constants::missing::doubleValue);
170  for (UInt f = 0; f < numFaces; ++f)
171  {
172  m_faceResults[f] = Interpolation(m_mesh.m_facesMassCenters[f]);
173  }
174  }

176  template <InterpolatableType T>
177  double BilinearInterpolationOnGriddedSamples<T>::Interpolation(const Point& point) const
178  {
179 
180  double fractionalColumnIndex = GetFractionalNumberOfColumns(point);
181  double fractionalRowIndex = GetFractionalNumberOfRows(point);
182 
183  double columnIndexTmp;
184  fractionalColumnIndex = std::modf(fractionalColumnIndex, &columnIndexTmp);
185 
186  double rowIndexTmp;
187  fractionalRowIndex = std::modf(fractionalRowIndex, &rowIndexTmp);
188 
189  if (columnIndexTmp < 0 || rowIndexTmp < 0)
190  {
191  return constants::missing::doubleValue;
192  }
193 
194  UInt const columnIndex = static_cast<UInt>(columnIndexTmp);
195  UInt const rowIndex = static_cast<UInt>(rowIndexTmp);
196 
197  if (columnIndex + 1 >= m_numXCoord || rowIndex + 1 >= m_numYCoord)
198  {
199  return constants::missing::doubleValue;
200  }
201 
202  const auto result = fractionalColumnIndex * fractionalRowIndex * GetGriddedValue(columnIndex + 1, rowIndex + 1) +
203  (1.0 - fractionalColumnIndex) * fractionalRowIndex * GetGriddedValue(columnIndex, rowIndex + 1) +
204  (1.0 - fractionalColumnIndex) * (1.0 - fractionalRowIndex) * GetGriddedValue(columnIndex, rowIndex) +
205  fractionalColumnIndex * (1.0 - fractionalRowIndex) * GetGriddedValue(columnIndex + 1, rowIndex);
206  return result;
207  }

209  template <InterpolatableType T>
210  double BilinearInterpolationOnGriddedSamples<T>::GetFractionalNumberOfColumns(const Point& point) const
211  {
212  if (m_isCellSizeConstant)
213  {
214  return (point.x - m_origin.x) / m_cellSize;
215  }
216  double result = constants::missing::doubleValue;
217  if (m_xCoordinates.size() < 2)
218  {
219  return result;
220  }
221  for (UInt i = 0; i < m_xCoordinates.size() - 1; ++i)
222  {
223 
224  if (point.x >= m_xCoordinates[i] && point.x < m_xCoordinates[i + 1])
225  {
226  const double dx = m_xCoordinates[i + 1] - m_xCoordinates[i];
227  result = static_cast<double>(i) + (point.x - m_xCoordinates[i]) / dx;
228  break;
229  }
230  }
231  return result;
232  }
