194  template <typename F>
195  [[nodiscard]] double FindFunctionRootWithGoldenSectionSearch(F func, double min, double max)
196  {
197  // golden distance factors
198  const double c = 0.38196602;
199  const double r = 0.61803399;
200  const double tolerance = 1e-5;
201 
202  double left = min;
203  double middle = (min + max) * 0.5;
204  double right = max;
205 
206  double x0 = left;
207  double x1 = middle - c * (middle - left);
208  double x2 = middle;
209  double x3 = right;
210  if (std::abs(right - middle) > std::abs(middle - left))
211  {
212  x1 = middle;
213  x2 = middle + c * (right - left);
214  }
215 
216  double f1 = func(x1);
217  double f2 = func(x2);
218 
219  while (std::abs(x3 - x0) > tolerance * std::max(std::abs(x1) + std::abs(x2), 1e-8)) // tolerance * std::max(std::abs(x1) + std::abs(x2), 1e-10)
220  {
221  if (f2 < f1)
222  {
223  x0 = x1;
224  x1 = x2;
225  x2 = r * x1 + c * x3;
226 
227  f1 = f2;
228  f2 = func(x2);
229  }
230  else
231  {
232  x3 = x2;
233  x2 = x1;
234  x1 = r * x2 + c * x0;
235 
236  f2 = f1;
237  f1 = func(x1);
238  }
239  }
240 
241  return f1 < f2 ? x1 : x2;
242  }

472  template <typename T>
473  [[nodiscard]] T ComputePointOnSplineAtAdimensionalDistance(const std::vector<T>& coordinates,
474  const std::vector<T>& coordinatesDerivatives,
475  double pointAdimensionalCoordinate)
476  {
477  T pointCoordinate{};
478  if (pointAdimensionalCoordinate < 0)
479  {
480  return pointCoordinate;
481  }
482 
483  const double eps = 1e-5;
484  const double splFac = 1.0;
485  const auto coordinate = std::floor(pointAdimensionalCoordinate);
486  if (pointAdimensionalCoordinate - coordinate < eps)
487  {
488  return pointCoordinate = coordinates[static_cast<UInt>(coordinate)];
489  }
490 
491  const UInt low = static_cast<UInt>(coordinate);
492  const UInt high = low + 1;
493  const double a = high - pointAdimensionalCoordinate;
494  const double b = pointAdimensionalCoordinate - low;
495 
496  pointCoordinate = coordinates[low] * a + coordinates[high] * b +
497  (coordinatesDerivatives[low] * (pow(a, 3) - a) + coordinatesDerivatives[high] * (pow(b, 3) - b)) / 6.0 * splFac;
498 
499  return pointCoordinate;
500  }
