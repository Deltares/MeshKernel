184  template <typename F>
185  [[nodiscard]] double FindFunctionRootWithGoldenSectionSearch(F func, double min, double max)
186  {
187  // golden distance factors
188  const double c = 0.38196602;
189  const double r = 0.61803399;
190  const double tolerance = 1e-5;
191 
192  double left = min;
193  double middle = (min + max) * 0.5;
194  double right = max;
195 
196  double x0 = left;
197  double x1 = middle - c * (middle - left);
198  double x2 = middle;
199  double x3 = right;
200  if (std::abs(right - middle) > std::abs(middle - left))
201  {
202  x1 = middle;
203  x2 = middle + c * (right - left);
204  }
205 
206  double f1 = func(x1);
207  double f2 = func(x2);
208 
209  while (std::abs(x3 - x0) > tolerance * std::max(std::abs(x1) + std::abs(x2), 1e-8)) // tolerance * std::max(std::abs(x1) + std::abs(x2), 1e-10)
210  {
211  if (f2 < f1)
212  {
213  x0 = x1;
214  x1 = x2;
215  x2 = r * x1 + c * x3;
216 
217  f1 = f2;
218  f2 = func(x2);
219  }
220  else
221  {
222  x3 = x2;
223  x2 = x1;
224  x1 = r * x2 + c * x0;
225 
226  f2 = f1;
227  f1 = func(x1);
228  }
229  }
230 
231  return f1 < f2 ? x1 : x2;
232  }

477  template <typename T>
478  [[nodiscard]] T ComputePointOnSplineAtAdimensionalDistance(const std::vector<T>& coordinates,
479  const std::vector<T>& coordinatesDerivatives,
480  double pointAdimensionalCoordinate)
481  {
482  T pointCoordinate{};
483  if (pointAdimensionalCoordinate < 0)
484  {
485  return pointCoordinate;
486  }
487 
488  const double eps = 1e-5;
489  const double splFac = 1.0;
490  const auto coordinate = std::floor(pointAdimensionalCoordinate);
491 
492  if (pointAdimensionalCoordinate - coordinate < eps)
493  {
494  return pointCoordinate = coordinates[static_cast<UInt>(coordinate)];
495  }
496 
497  const UInt low = static_cast<UInt>(coordinate);
498  const UInt high = low + 1;
499  const double a = high - pointAdimensionalCoordinate;
500  const double b = pointAdimensionalCoordinate - low;
501 
502  pointCoordinate = coordinates[low] * a + coordinates[high] * b +
503  (coordinatesDerivatives[low] * (pow(a, 3) - a) + coordinatesDerivatives[high] * (pow(b, 3) - b)) / 6.0 * splFac;
504 
505  return pointCoordinate;
506  }
