218  template <typename F>
219  [[nodiscard]] double FindFunctionRootWithGoldenSectionSearch(F func, double min, double max)
220  {
221  // golden distance factors
222  const double c = 0.38196602;
223  const double r = 0.61803399;
224  const double tolerance = 1e-5;
225 
226  double left = min;
227  double middle = (min + max) * 0.5;
228  double right = max;
229 
230  double x0 = left;
231  double x1 = middle - c * (middle - left);
232  double x2 = middle;
233  double x3 = right;
234  if (std::abs(right - middle) > std::abs(middle - left))
235  {
236  x1 = middle;
237  x2 = middle + c * (right - left);
238  }
239 
240  double f1 = func(x1);
241  double f2 = func(x2);
242 
243  while (std::abs(x3 - x0) > tolerance * std::max(std::abs(x1) + std::abs(x2), 1e-8)) // tolerance * std::max(std::abs(x1) + std::abs(x2), 1e-10)
244  {
245  if (f2 < f1)
246  {
247  x0 = x1;
248  x1 = x2;
249  x2 = r * x1 + c * x3;
250 
251  f1 = f2;
252  f2 = func(x2);
253  }
254  else
255  {
256  x3 = x2;
257  x2 = x1;
258  x1 = r * x2 + c * x0;
259 
260  f2 = f1;
261  f1 = func(x1);
262  }
263  }
264 
265  return f1 < f2 ? x1 : x2;
266  }

496  template <typename T>
497  [[nodiscard]] T ComputePointOnSplineAtAdimensionalDistance(const std::vector<T>& coordinates,
498  const std::vector<T>& coordinatesDerivatives,
499  double pointAdimensionalCoordinate)
500  {
501  T pointCoordinate{};
502  if (pointAdimensionalCoordinate < 0)
503  {
504  return pointCoordinate;
505  }
506 
507  const double eps = 1e-5;
508  const double splFac = 1.0;
509  const auto coordinate = std::floor(pointAdimensionalCoordinate);
510 
511  if (pointAdimensionalCoordinate - coordinate < eps)
512  {
513  return pointCoordinate = coordinates[static_cast<UInt>(coordinate)];
514  }
515 
516  const UInt low = static_cast<UInt>(coordinate);
517  const UInt high = low + 1;
518  const double a = high - pointAdimensionalCoordinate;
519  const double b = pointAdimensionalCoordinate - low;
520 
521  pointCoordinate = coordinates[low] * a + coordinates[high] * b +
522  (coordinatesDerivatives[low] * (pow(a, 3) - a) + coordinatesDerivatives[high] * (pow(b, 3) - b)) / 6.0 * splFac;
523 
524  return pointCoordinate;
525  }
